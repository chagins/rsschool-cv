// Roman to Integer
// https://leetcode.com/problems/roman-to-integer/
/**
  * @param {string} s
  * @return {number}
  */
let romanToInt = function (s) {
  const ROMAN = new Map([
    ["I", 1],
    ["V", 5],
    ["X", 10],
    ["L", 50],
    ["C", 100],
    ["D", 500],
    ["M", 1000],
  ]);

  let date = 0;
  for (let i = 0; i < s.length; i++) {

    const currentNum = ROMAN.get(s[i]);
    const nextNum = ROMAN.get(s[i + 1]);

    date += 
      (i < s.length - 1 && currentNum < nextNum) 
      ? -currentNum 
      : currentNum;
  }

  return date;
};
